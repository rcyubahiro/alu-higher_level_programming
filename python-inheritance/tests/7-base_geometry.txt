from base_geometry import BaseGeometry

# Test instantiation
bg = BaseGeometry()

# Test area() method
try:
    bg.area()
except Exception as e:
    print(e)  # Expected output: "area() is not implemented"

# Test integer_validator() method
try:
    bg.integer_validator("my_int", 12)  # Should pass without error
    print("integer_validator passed for 12")
except Exception as e:
    print(e)

# Test integer_validator with invalid values
test_cases = [
    ("age", 1),       # Valid
    ("age", 0),       # Should raise ValueError
    ("age", -4),      # Should raise ValueError
    ("age", "4"),     # Should raise TypeError
    ("age", (4,)),    # Should raise TypeError
    ("age", [3]),     # Should raise TypeError
    ("age", True),    # Should raise TypeError
    ("age", {3, 4}),  # Should raise TypeError
    ("age", None)     # Should raise TypeError
]

for name, value in test_cases:
    try:
        bg.integer_validator(name, value)
        print(f"integer_validator passed for {value}")
    except Exception as e:
        print(e)
